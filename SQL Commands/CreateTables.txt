CREATE TABLE Patient (
PatientID INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
PatientFName VARCHAR(100) NOT NULL,
PatientSName VARCHAR(100) NOT NULL,
PatientEmail VARCHAR(255) NOT NULL,
PatientPwd VARCHAR(30) NOT NULL,
PatientPostcode VARCHAR(10),
PatientCode VARCHAR(10) NOT NULL
);

CREATE TABLE Carer (
CarerID INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
CarerFName VARCHAR(100) NOT NULL,
CarerSName VARCHAR(100) NOT NULL,
CarerEmail VARCHAR(255) NOT NULL,
CarerPwd VARCHAR(30) NOT NULL
);

CREATE TABLE Location (
LocationID INT PRIMARY KEY IDENTITY(1,1) NOT NULL,
PatientID INT FOREIGN KEY REFERENCES Patient(PatientID) NOT NULL,
Longitude FLOAT NOT NULL,
Latitude FLOAT NOT NULL,
TimeLocated DATETIME DEFAULT GETDATE()
);

CREATE TABLE CarerPatient (
CarerID INT FOREIGN KEY REFERENCES Carer(CarerID) NOT NULL,
PatientID INT FOREIGN KEY REFERENCES Patient(PatientID) NOT NULL,
CONSTRAINT cp_PK PRIMARY KEY (CarerID, PatientID) 
);
